# Custom Dropdown

```jsx
import { Main, DropdownContainer, DropdownHeader, DropdownList, DropdownListContainer, ListItem, Backdrop } from './src/stories/Dropdown';

const CustomDropdown: React.FC<TDropdownProps> = ({ // Customize default options
    options, label, isSelect, avatar = "https://github.com/hugoleonardodev.png", shape = 'rounded', hasShadow = false
  }) => {
  // Customize states
  const [isOpen, setIsOpen] = React.useState(false);
  const [selectedOption, setSelectedOption] = React.useState(0);

  // Customize toggling
  const toggling = () => {
    setIsOpen(!isOpen);
  };

  // Customize onOptionClicked
  const onOptionClicked = (value: number) => () => {
    setSelectedOption(value);
    setIsOpen(false);
  };

  // Customize rendering options
  return (
    <Main>
      <DropdownContainer>
        <DropdownHeader onClick={toggling} shape={shape} hasShadow={hasShadow}>
          {avatar && <Chip avatar={avatar} size="small"/>}
          {isSelect ? (selectedOption || label) : label}
          <CaretIcon />
        </DropdownHeader>
        {isOpen && (
          <DropdownListContainer>
            <DropdownList>
              {options.map((option) => (
                <ListItem
                  onClick={onOptionClicked(option.value)}
                  key={option.value}
                >
                  {option.label}
                </ListItem>
              ))}
            </DropdownList>
          </DropdownListContainer>
        )}
      </DropdownContainer>
      <Backdrop isOpen={isOpen} onClick={toggling} />
    </Main>
  );
};

export default CustomDropdown;
```
